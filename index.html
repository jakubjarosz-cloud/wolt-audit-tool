    <script>

        const apiKey = "TWÓJ_KLUCZ_API"; // Wklej swój klucz tutaj

        const model = "gemini-2.0-flash";

        let menuChart;


        const systemInstruction = "Jesteś Specjalistą Onboardingu w Wolt. Twoim zadaniem jest analiza linków restauracji i przygotowanie danych do systemu. Twój styl jest profesjonalny, nowoczesny i zorientowany na sprzedaż (foodie-friendly).";


        // Konfiguracja Schematu JSON

        const responseSchema = {

            type: "object",

            properties: {

                name: { type: "string" },

                address: { type: "string" },

                slogan: { type: "string" },

                description: { type: "string" },

                min_basket: { type: "number" },

                own_delivery: { type: "string", enum: ["TAK", "NIE", "NIEZNANY"] },

                hours: {

                    type: "object",

                    properties: {

                        Pn: { type: "string" }, Wt: { type: "string" }, Sr: { type: "string" },

                        Cz: { type: "string" }, Pt: { type: "string" }, So: { type: "string" }, Nd: { type: "string" }

                    }

                },

                menu_stats: { type: "array", items: { type: "number" } }

            },

            required: ["name", "address", "hours", "menu_stats"]

        };


        async function fetchAI(prompt, useSchema = true) {

            const url = `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${apiKey}`;

            

            const payload = {

                contents: [{ parts: [{ text: prompt }] }],

                system_instruction: { parts: [{ text: systemInstruction }] },

                tools: [{ "google_search": {} }],

                generationConfig: {

                    temperature: 0.7,

                    ...(useSchema && {

                        response_mime_type: "application/json",

                        response_schema: responseSchema

                    })

                }

            };


            try {

                const response = await fetch(url, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

                const data = await response.json();

                return data.candidates[0].content.parts[0].text;

            } catch (e) {

                console.error("Błąd API:", e);

                return null;

            }

        }


        async function triggerFullAnalysis() {

            const url = document.getElementById('merchantUrl').value;

            if (!url) return alert("Podaj URL!");


            document.getElementById('loader').style.display = 'flex';


            const prompt = `Analizuj link: ${url}. Wyciągnij dane lokalu. Slogan i opis napisz po polsku w stylu Wolt. Menu stats: [Dania główne, Dodatki, Napoje, Desery].`;

            

            const result = await fetchAI(prompt, true);

            if (result) {

                const data = JSON.parse(result);

                updateUI(data);

            }

            document.getElementById('loader').style.display = 'none';

        }


        function updateUI(data) {

            document.getElementById('v_name').innerText = data.name;

            document.getElementById('v_address').innerText = data.address;

            document.getElementById('v_slogan').innerText = `"${data.slogan || 'Świeżo i lokalnie'}"`;

            document.getElementById('v_desc').innerText = data.description || 'Brak opisu.';

            document.getElementById('v_min_basket').innerHTML = `${data.min_basket || '--'} <small class="text-lg opacity-30">PLN</small>`;

            document.getElementById('v_own_delivery').innerText = data.own_delivery;


            const days = { Pn: "Poniedziałek", Wt: "Wtorek", Sr: "Środa", Cz: "Czwartek", Pt: "Piątek", So: "Sobota", Nd: "Niedziela" };

            const tableBody = document.getElementById('v_hours_table');

            tableBody.innerHTML = '';

            

            Object.keys(days).forEach(key => {

                tableBody.innerHTML += `

                    <tr class="table-row">

                        <td class="px-6 py-4 font-bold text-sm">${days[key]}</td>

                        <td class="px-6 py-4 font-black text-wolt-blue text-right">${data.hours[key] || '---'}</td>

                    </tr>

                `;

            });


            updateChart(data.menu_stats || [10, 5, 5, 2]);

        }


        function updateChart(stats) {

            if (menuChart) menuChart.destroy();

            const ctx = document.getElementById('menuChart').getContext('2d');

            menuChart = new Chart(ctx, {

                type: 'doughnut',

                data: {

                    labels: ['Główne', 'Dodatki', 'Napoje', 'Desery'],

                    datasets: [{

                        data: stats,

                        backgroundColor: ['#009DE0', '#00C2E8', '#1A2B49', '#E5E7EB'],

                        borderWidth: 5

                    }]

                },

                options: { maintainAspectRatio: false, cutout: '75%' }

            });

        }


        async function generateAuditReport() {

            const area = document.getElementById('finalReportArea');

            const name = document.getElementById('v_name').innerText;

            if (name === "---") return alert("Najpierw wykonaj analizę!");


            area.innerText = "⏳ Generowanie oficjalnego werdyktu...";

            

            const prompt = `Napisz notatkę systemową Wolt (Internal Memo) dla restauracji ${name}. 

            Struktura: [POTENCJAŁ RYNKOWY (1-10)], [ANALIZA OPERACYJNA], [REKOMENDACJE], [WERDYKT]. 

            Styl: Profesjonalny raport operacyjny.`;


            const res = await fetchAI(prompt, false);

            area.innerText = res;

        }


        window.onload = () => updateChart([0, 0, 0, 0]);
